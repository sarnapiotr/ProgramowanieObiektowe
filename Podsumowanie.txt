Interfejs - Same deklaracje, bez implementacji

public interface Figura {
	public float obliczObwod();
	public float obliczPowierzchnie();
}

class Kwadrat implements Figura {
	private float a;

    public Kwadrat(float a) {
        this.a = a;
    }

	@Override
    public float obliczObwod() {
    	return 4 * a;
	}

   	 @Override
    public float obliczPowierzchnie() {
    	return a * a;
    }
}

Polimorfizm

public class Punkt2D {
	protected float x, y;

	public Punkt2D(float x, float y) {
		this.x = x;
		this.y = y;
	}
}

public class Punkt3D extends Punkt2D {
	protected float z;

	public Punkt3D(float x, float y, float z) {
		super(x, y);
		this.z = z;
	}
}

Punkt2D punkt2D = new Punkt2D(10, 20);
Punkt2D punkt3D = new Punkt3D(10, 20, 30);
